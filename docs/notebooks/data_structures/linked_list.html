    class LinkedList(object):
    def __init__(self):
        self.head = None
        self.tail = None
        self.counter = 0

    def __iter__(self):
        current = self.head
        while current:
            print(current.data)
            val = current.data
            current = current.next
            yield val

    def push_front(self, data):
        if data is None:
            return False
        node = Node(data)
        current = self.head
        if self.head:
            while current.next:
                current = current.next
            current.next = node
        else:
            self.head = node
            self.tail = node
        self.head = node
        self.counter += 1

    def push_back(self, data):
        node = Node(data)
        current = self.head
        if current is None:
            current.head = node
        else:
            while current.next is not None:
                current = current.next
            current.next = node
